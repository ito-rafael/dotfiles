** Header :noexport_0:
*** Template
#+name: install
#+begin_src elisp :var pin=0 comma=1 extension_id="aaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaa" slug="default-slug" :results output :tangle no
(princ (format "\"%s\": {\n" extension_id))
(when (= pin 1)
  (princ "  \"toolbar_pin\": \"force_pinned\",\n"))
(princ "  \"installation_mode\": \"normal_installed\",\n")
(princ (format "  \"install_url\": \"https://addons.mozilla.org/firefox/downloads/latest/%s/latest.xpi\"\n" slug))
(if (= comma 1)
    (princ "},\n")
  (princ "}\n"))
#+end_src
