    echo -e "$@"

#+begin_src bash
show_help() {
  echo "Usage: $0 <trigger|status|logs|run> [options] [Task ID]"
  echo "Options:"
  echo "  -h, --help        Show this help message"
  echo "  -v, --verbose     Show detailed execution messages"
  echo "  --test            Use the 'local (test)' template"
  echo "  --linear          Run using the Linear strategy (bypasses Mitogen)"
  echo "  --tags TAGS       Specify tags to run (eg: 'waybar,kanata')"
  echo "  --skip-tags TAGS  Specify tags to skip"
  echo ""
  echo "Examples:"
  echo "  $0 run --test --tags \"acl,users\" -v"
  echo "  $0 run --skip-tags \"emacs\""
  echo "  $0 run --test --linear -v"
  echo "  $0 status 42"
}
#+end_src
** Parse parameters
#+begin_src bash
COMMAND="$1"
shift  # remove the command from the argument list so we can loop over the rest

# if help flags or no command
if [ -z "$COMMAND" ] || [[ "$COMMAND" == "-h" ]] || [[ "$COMMAND" == "--help" ]]; then
  show_help
  exit 1
fi

# set defaults
TEMPLATE_NAME="ansible-provision"
TASK_ID=""
TAGS=""
SKIP_TAGS=""
VERBOSE_FLAG=""
LINEAR=false

# loop through remaining arguments
while [[ "$#" -gt 0 ]]; do
  case $1 in
    -h|--help)
      show_help
      exit 0
      ;;
    -v|--verbose)
      VERBOSE=true
      VERBOSE_FLAG="-v"
      shift
      ;;
    --test)
      TEMPLATE_NAME="local (test)"
      shift
      ;;
    --linear)
      LINEAR=true
      shift
      ;;
    --tags)
      # remove all spaces
      TAGS="${2// /}"
      shift 2
      ;;
    --skip-tags)
      # remove all spaces
      SKIP_TAGS="${2// /}"
      shift 2
      ;;
    ,*)
      # if it is not a recognized flag, we assume it is the Task ID
      if [[ "$1" != -* ]]; then
        TASK_ID="$1"
      fi
      shift
      ;;
  esac
done

# append " [linear]" to the Template Name if the flag was passed
if [ "$LINEAR" = true ]; then
    TEMPLATE_NAME="${TEMPLATE_NAME} [linear]"
fi

# safety check: prevent running status/logs without an ID
if [[ "$COMMAND" == "status" || "$COMMAND" == "logs" ]] && [ -z "$TASK_ID" ]; then
  echo "Error: You must provide a Task ID for the '$COMMAND' command." >&2
  echo "Example: $0 $COMMAND 42" >&2
  exit 1
fi
#+end_src

#+begin_src bash
"trigger")
    vecho "Launching Semaphore Task: $TEMPLATE_NAME..."

    # build the base JSON payload safely
    JSON_PAYLOAD=$(jq -n \
      --arg t_id "$TEMPLATE_ID" \
      --arg msg "$COMMIT_MESSAGE" \
      '{template_id: ($t_id | tonumber), message: $msg}')

    # inject "tags" inside the params object as a JSON array
    if [ -n "$TAGS" ]; then
      JSON_PAYLOAD=$(echo "$JSON_PAYLOAD" | jq --arg tags "$TAGS" '.params.tags = ($tags | split(","))')
    fi

    # inject "skip-tags" inside the params object as a JSON array
    if [ -n "$SKIP_TAGS" ]; then
      JSON_PAYLOAD=$(echo "$JSON_PAYLOAD" | jq --arg skips "$SKIP_TAGS" '.params.skip_tags = ($skips | split(","))')
    fi

    RESPONSE=$(curl -s -X POST -H "Authorization: Bearer $TOKEN" -H "Content-Type: application/json" \
        -d "$JSON_PAYLOAD" \
        "$API_URL/project/$PROJECT_ID/tasks")

    TASK_ID=$(echo "$RESPONSE" | jq -r '.id')
    
    if [ -n "$TASK_ID" ] && [ "$TASK_ID" != "null" ]; then
      vecho "Success! Task queued with ID: $TASK_ID"
    else
      echo "Failed to launch task. API Response:" >&2
      echo "$RESPONSE" >&2
    fi
    ;;
#+end_src
** Get status
#+begin_src bash
"status")
    STATUS=$(curl -s -H "Authorization: Bearer $TOKEN" "$API_URL/project/$PROJECT_ID/tasks/$TASK_ID" | jq -r '.status')
    if [ -z "$STATUS" ] || [ "$STATUS" == "null" ]; then
      echo "Error: Task ID $TASK_ID not found." >&2
      exit 1
    fi
    echo "$STATUS"
    ;;
#+end_src
** Get logs
#+begin_src bash
"logs")
    # verify the task exists before trying to loop
    STATUS=$(curl -s -H "Authorization: Bearer $TOKEN" "$API_URL/project/$PROJECT_ID/tasks/$TASK_ID" | jq -r '.status')
    if [ -z "$STATUS" ] || [ "$STATUS" == "null" ]; then
      echo "Error: Task ID $TASK_ID not found." >&2
      exit 1
    fi

    vecho "Tailing logs for Task ID: $TASK_ID..."
    vecho "Press Ctrl+C at any time to cancel this task."
    vecho "----------------------------------------"

    # trap Ctrl+C (SIGINT) to send the stop API call before exiting
    trap '
        vecho "\n\n[!] Ctrl+C detected. Sending stop signal to Semaphore for Task $TASK_ID..."

        # Send an empty JSON object to satisfy strict routers, and capture the HTTP status code
        STOP_RESPONSE=$(curl -s -w "\nHTTP_STATUS: %{http_code}" -X POST -H "Authorization: Bearer $TOKEN" -H "Content-Type: application/json" -d "{}" "$API_URL/project/$PROJECT_ID/tasks/$TASK_ID/stop")

        vecho "API Response:\n$STOP_RESPONSE"
        vecho "Task cancellation requested. Exiting terminal."
        exit 130
    ' SIGINT

    LAST_INDEX=0
    # enter the live tail loop
    while [ "$STATUS" == "waiting" ] || [ "$STATUS" == "running" ]; do
        # fetch the full log array
        RAW_OUTPUT=$(curl -s -H "Authorization: Bearer $TOKEN" "$API_URL/project/$PROJECT_ID/tasks/$TASK_ID/output")
        if [ "$RAW_OUTPUT" != "null" ] && [ -n "$RAW_OUTPUT" ]; then
            # print only new log entries starting from LAST_INDEX
            echo "$RAW_OUTPUT" | jq -r ".[$LAST_INDEX:] | .[]?.output"
            # update LAST_INDEX to the new total length of the array
            LAST_INDEX=$(echo "$RAW_OUTPUT" | jq length)
        fi

        # update the status to see if we should keep looping
        STATUS=$(curl -s -H "Authorization: Bearer $TOKEN" "$API_URL/project/$PROJECT_ID/tasks/$TASK_ID" | jq -r '.status')
        if [ "$STATUS" != "waiting" ] && [ "$STATUS" != "running" ]; then
            break
        fi

        # sleep for 2 seconds to avoid hammering the Semaphore API
        sleep 2
    done

    # remove the trap cleanly if the task finishes on its own
    trap - SIGINT

    vecho "----------------------------------------"
    vecho "Task finished with status: $STATUS"
    ;;
#+end_src
** Run (Trigger and Watch)
#+begin_src bash
"run")
    vecho "Launching Semaphore Task: $TEMPLATE_NAME..."

    # build the base JSON payload safely
    JSON_PAYLOAD=$(jq -n \
      --arg t_id "$TEMPLATE_ID" \
      --arg msg "$COMMIT_MESSAGE" \
      '{template_id: ($t_id | tonumber), message: $msg}')

    # inject "tags" inside the params object as a JSON array
    if [ -n "$TAGS" ]; then
      JSON_PAYLOAD=$(echo "$JSON_PAYLOAD" | jq --arg tags "$TAGS" '.params.tags = ($tags | split(","))')
    fi

    # inject "skip-tags" inside the params object as a JSON array
    if [ -n "$SKIP_TAGS" ]; then
      JSON_PAYLOAD=$(echo "$JSON_PAYLOAD" | jq --arg skips "$SKIP_TAGS" '.params.skip_tags = ($skips | split(","))')
    fi

    RESPONSE=$(curl -s -X POST -H "Authorization: Bearer $TOKEN" -H "Content-Type: application/json" \
        -d "$JSON_PAYLOAD" \
        "$API_URL/project/$PROJECT_ID/tasks")

    TASK_ID=$(echo "$RESPONSE" | jq -r '.id')

    if [ -n "$TASK_ID" ] && [ "$TASK_ID" != "null" ]; then
      vecho "Success! Task queued with ID: $TASK_ID"
      vecho "Switching to live logs..."
      vecho "----------------------------------------"

      # recursively call this exact same script using the 'logs' command, forwarding the verbose flag if set
      "$0" logs "$TASK_ID" $VERBOSE_FLAG

    else
      echo "Failed to launch task. API Response:" >&2
      echo "$RESPONSE" >&2
    fi
    ;;
#+end_src
** EoS
#+begin_src bash
*)
    echo "Invalid command: $COMMAND" >&2
    echo "Available commands: trigger, status, logs, run" >&2
    exit 1
    ;;
esac
