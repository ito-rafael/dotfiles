TOKEN=$(pass show semaphore/opentofu_token)
if [ -z "$TOKEN" ]; then
  echo "Error: No token provided in environment."
  exit 1

** Parse parameters
#+begin_src bash
COMMAND="$1"
shift  # remove the command from the argument list so we can loop over the rest

if [ -z "$COMMAND" ]; then
  echo "Usage: $0 <trigger|status|logs|run> [options] [Task ID]"
  echo "Options:"
  echo "  --test            Use the 'local (test)' template"
  echo "  --tags TAGS       Specify tags to run (e.g., 'nginx,postgres')"
  echo "  --skip-tags TAGS  Specify tags to skip"
  echo ""
  echo "Examples:"
  echo "  $0 run --test --tags \"acl,users\""
  echo "  $0 run --skip-tags \"emacs\""
  echo "  $0 status 42"
  exit 1
fi

# set defaults
TEMPLATE_NAME="ansible-provision"
TASK_ID=""
TAGS=""
SKIP_TAGS=""

# loop through remaining arguments
while [[ "$#" -gt 0 ]]; do
  case $1 in
    --test)
      TEMPLATE_NAME="local (test)"
      shift
      ;;
    --tags)
      TAGS="$2"
      shift 2
      ;;
    --skip-tags)
      SKIP_TAGS="$2"
      shift 2
      ;;
    *)
      # if it is not a recognized flag, we assume it is the Task ID
      if [[ "$1" != -* ]]; then
        TASK_ID="$1"
      fi
      shift
      ;;
  esac
done

# safety check: prevent running status/logs without an ID
if [[ "$COMMAND" == "status" || "$COMMAND" == "logs" ]] && [ -z "$TASK_ID" ]; then
  echo "Error: You must provide a Task ID for the '$COMMAND' command."
  echo "Example: $0 $COMMAND 42"
  exit 1
fi
#+end_src
* Gather common info
** Get project ID
#+begin_src bash
PROJECT_ID=$(curl -s -H "Authorization: Bearer $TOKEN" "$API_URL/projects" | jq -r ".[] | select(.name == \"$PROJECT_NAME\") | .id")
if [ -z "$PROJECT_ID" ] || [ "$PROJECT_ID" == "null" ]; then
  echo "Error: Project '$PROJECT_NAME' not found."
  exit 1
fi
#+end_src
** Get Template ID
#+begin_src bash
if [ "$COMMAND" == "trigger" ] || [ "$COMMAND" == "run" ]; then
  TEMPLATE_ID=$(curl -s -H "Authorization: Bearer $TOKEN" "$API_URL/project/$PROJECT_ID/templates" | jq -r ".[] | select(.name == \"$TEMPLATE_NAME\") | .id")
  
  if [ -z "$TEMPLATE_ID" ] || [ "$TEMPLATE_ID" == "null" ]; then
    echo "Error: Template '$TEMPLATE_NAME' not found."
    exit 1
  fi
fi
#+end_src

#+begin_src bash
case "$COMMAND" in
#+end_src
** Trigger task
#+begin_src bash
"trigger")
    echo "Launching Semaphore Task: $TEMPLATE_NAME..."

    # safely build the base JSON payload using jq
    JSON_PAYLOAD=$(jq -n \
      --arg t_id "$TEMPLATE_ID" \
      --arg msg "$COMMIT_MESSAGE" \
      '{template_id: ($t_id | tonumber), message: $msg}')

    # inject "tags" into the params object as an array
    if [ -n "$TAGS" ]; then
      JSON_PAYLOAD=$(echo "$JSON_PAYLOAD" | jq \
        --arg tags "$TAGS" \
        '.params.tags = ($tags | split(","))')
    fi

    # inject "skip-tags" into the params object as an array
    if [ -n "$SKIP_TAGS" ]; then
      JSON_PAYLOAD=$(echo "$JSON_PAYLOAD" | jq \
        --arg skips "$SKIP_TAGS" \
        '.params.skip_tags = ($skips | split(","))')
    fi

    RESPONSE=$(curl -s -X POST -H "Authorization: Bearer $TOKEN" -H "Content-Type: application/json" \
        -d "$JSON_PAYLOAD" \
        "$API_URL/project/$PROJECT_ID/tasks")

    TASK_ID=$(echo "$RESPONSE" | jq -r '.id')
    
    if [ -n "$TASK_ID" ] && [ "$TASK_ID" != "null" ]; then
      echo "Success! Task queued with ID: $TASK_ID"
    else
      echo "Failed to launch task. API Response:"
      echo "$RESPONSE"
    fi
    ;;
#+end_src
** Get status
#+begin_src bash
"status")
    STATUS=$(curl -s -H "Authorization: Bearer $TOKEN" "$API_URL/project/$PROJECT_ID/tasks/$TASK_ID" | jq -r '.status')
    if [ -z "$STATUS" ] || [ "$STATUS" == "null" ]; then
      echo "Error: Task ID $TASK_ID not found."
      exit 1
    fi
    echo "$STATUS"
    ;;
#+end_src
** Get logs
#+begin_src bash
"logs")
    # verify the task exists before trying to loop
    STATUS=$(curl -s -H "Authorization: Bearer $TOKEN" "$API_URL/project/$PROJECT_ID/tasks/$TASK_ID" | jq -r '.status')
    if [ -z "$STATUS" ] || [ "$STATUS" == "null" ]; then
      echo "Error: Task ID $TASK_ID not found."
      exit 1
    fi

    LAST_INDEX=0
    # enter the live tail loop
    while [ "$STATUS" == "waiting" ] || [ "$STATUS" == "running" ]; do
        # fetch the full log array
        RAW_OUTPUT=$(curl -s -H "Authorization: Bearer $TOKEN" "$API_URL/project/$PROJECT_ID/tasks/$TASK_ID/output")
        if [ "$RAW_OUTPUT" != "null" ] && [ -n "$RAW_OUTPUT" ]; then
            # print only new log entries starting from LAST_INDEX
            echo "$RAW_OUTPUT" | jq -r ".[$LAST_INDEX:] | .[]?.output"
            # update LAST_INDEX to the new total length of the array
            LAST_INDEX=$(echo "$RAW_OUTPUT" | jq length)
        fi

        # update the status to see if we should keep looping
        STATUS=$(curl -s -H "Authorization: Bearer $TOKEN" "$API_URL/project/$PROJECT_ID/tasks/$TASK_ID" | jq -r '.status')
        if [ "$STATUS" != "waiting" ] && [ "$STATUS" != "running" ]; then
            break
        fi

        # sleep for 2 seconds to avoid hammering the Semaphore API
        sleep 2
    done
    ;;
#+end_src
** Run (Trigger and Watch)
#+begin_src bash
"run")
    echo "Launching Semaphore Task: $TEMPLATE_NAME..."

    # safely build the base JSON payload using jq
    JSON_PAYLOAD=$(jq -n \
      --arg t_id "$TEMPLATE_ID" \
      --arg msg "$COMMIT_MESSAGE" \
      '{template_id: ($t_id | tonumber), message: $msg}')

    # inject "tags" into the params object as an array
    if [ -n "$TAGS" ]; then
      JSON_PAYLOAD=$(echo "$JSON_PAYLOAD" | jq \
        --arg tags "$TAGS" \
        '.params.tags = ($tags | split(","))')
    fi

    # inject "skip-tags" into the params object as an array
    if [ -n "$SKIP_TAGS" ]; then
      JSON_PAYLOAD=$(echo "$JSON_PAYLOAD" | jq \
        --arg skips "$SKIP_TAGS" \
        '.params.skip_tags = ($skips | split(","))')
    fi

    RESPONSE=$(curl -s -X POST -H "Authorization: Bearer $TOKEN" -H "Content-Type: application/json" \
        -d "$JSON_PAYLOAD" \
        "$API_URL/project/$PROJECT_ID/tasks")

    TASK_ID=$(echo "$RESPONSE" | jq -r '.id')

    if [ -n "$TASK_ID" ] && [ "$TASK_ID" != "null" ]; then
      echo "Success! Task queued with ID: $TASK_ID"
      echo "Switching to live logs..."
      echo "----------------------------------------"

      # recursively call this exact same script using the 'logs' command
      "$0" logs "$TASK_ID"

    else
      echo "Failed to launch task. API Response:"
      echo "$RESPONSE"
    fi
    ;;
#+end_src
