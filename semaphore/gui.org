    def update_command(self, *args):
        """Dynamically builds the bash command based on UI state."""
        # cleaned up to output purely the arguments, so they drop cleanly into the kitty script
        cmd = ["run"]

        if self.strategy_drop.get_selected_item().get_string() == "Linear":
            cmd.append("--linear")

        if self.dry_run_check.get_active():
            cmd.append("--test")
        if self.diff_check.get_active():
            cmd.append("--diff")
        if self.debug_check.get_active():
            cmd.append("-v")

        tags = self.tags_entry.get_text().strip()
        if tags:
            cmd.append(f'--tags "{tags}"')

        skip_tags = self.skip_tags_entry.get_text().strip()
        if skip_tags:
            cmd.append(f'--skip-tags "{skip_tags}"')

        start_task = self.start_task_entry.get_text().strip()
        if start_task:
            cmd.append(f'--start-at-task "{start_task}"')

        self.command_view.get_buffer().set_text(" ".join(cmd))

#+begin_src python
    def get_nav_node(self, focus_widget):
        """Resolves internal GTK composite widgets (like Gtk.Text) to their parent UI node."""
        if not focus_widget:
            return None
        current = focus_widget
        while current:
            if hasattr(self, 'nav_map') and current in self.nav_map:
                return current
            current = current.get_parent()
        return None

    def set_mode(self, mode):
        """Explicitly handles the transition between NORMAL and INSERT."""
        self.current_mode = mode
        if hasattr(self, 'mode_label'):
            self.mode_label.set_text(mode)

        is_insert = (mode == "INSERT")

        # definitively disable GTK Input Methods to prevent leaked typing in NORMAL mode
        if hasattr(self, 'tags_entry'):
            self.tags_entry.set_editable(is_insert)
            self.skip_tags_entry.set_editable(is_insert)
            self.start_task_entry.set_editable(is_insert)

            # dynamically inject/remove the transparent caret CSS
            widgets_with_carets = [self.tags_entry, self.skip_tags_entry, self.start_task_entry, self.command_view]
            for w in widgets_with_carets:
                if is_insert:
                    w.remove_css_class("hide-caret")
                else:
                    w.add_css_class("hide-caret")

    def set_initial_focus(self):
        """Ensures the app starts cleanly focused on the tags entry in INSERT mode."""
        self.ignore_focus_change = True
        self.tags_entry.grab_focus()
        self.set_mode("INSERT")
        self.tags_entry.set_position(-1)
        self.ignore_focus_change = False
        return False # Removes the idle hook

    def on_focus_changed(self, *args):
        """Only triggered when the user explicitly CLICKS a widget or uses native Tab."""
        if self.ignore_focus_change:
            return

        focus_widget = self.get_focus()
        nav_node = self.get_nav_node(focus_widget)

        if isinstance(nav_node, Gtk.Entry):
            self.set_mode("INSERT")
        else:
            self.set_mode("NORMAL")

    def on_key_pressed(self, controller, keyval, keycode, state):
        """Handles Vim-style mode switching and Colemak-DH navigation."""
        focus_widget = self.get_focus()
        nav_node = self.get_nav_node(focus_widget)
        is_entry = isinstance(nav_node, Gtk.Entry)

        unicode_char = Gdk.keyval_to_unicode(keyval)
        key = chr(unicode_char) if unicode_char else ""

        # global Ctrl+C intercept to terminate the window
        if (state & Gdk.ModifierType.CONTROL_MASK) and keyval in (Gdk.KEY_c, Gdk.KEY_C):
            self.close()
            return True

        if keyval == Gdk.KEY_Escape:
            if self.current_mode == "INSERT":
                self.set_mode("NORMAL")
            else:
                self.close()
            return True

        if keyval in (Gdk.KEY_Return, Gdk.KEY_KP_Enter):
            # Enter launches the script and closes the window
            self.on_launch_clicked(self.launch_btn)
            return True

        # if in INSERT mode, let the keystrokes (including arrows) pass through cleanly
        if self.current_mode == "INSERT":
            return False

        # block non-printable modifiers from deleting text in NORMAL mode
        if is_entry and keyval in (Gdk.KEY_BackSpace, Gdk.KEY_Delete):
            return True

        # trigger INSERT mode
        if key == 's':
            if is_entry:
                self.set_mode("INSERT")
                nav_node.set_position(-1)
            return True

        # spatial navigation: Colemak-DH {neio} + arrow keys
        direction = None
        if keyval == Gdk.KEY_Left or key == 'n':
            direction = 'n'
        elif keyval == Gdk.KEY_Down or key == 'e':
            direction = 'e'
        elif keyval == Gdk.KEY_Up or key == 'i':
            direction = 'i'
        elif keyval == Gdk.KEY_Right or key == 'o':
            direction = 'o'

        if direction:
            if nav_node and nav_node in self.nav_map:
                target = self.nav_map[nav_node][direction]
                if target:
                    self.ignore_focus_change = True
                    target.grab_focus()
                    self.ignore_focus_change = False
            return True

        # let other structural keys (eg: Tab) act natively if not explicitly handled
        if not key:
            return False

        # direct jumps
        if key == 'h':
            self.ignore_focus_change = True
            self.hosts_drop.grab_focus()
            self.ignore_focus_change = False
            return True
        elif key == 'm':
            self.ignore_focus_change = True
            self.strategy_drop.grab_focus()
            self.ignore_focus_change = False
            return True
        elif key == 't':
            self.ignore_focus_change = True
            self.tags_entry.grab_focus()
            self.set_mode("INSERT") # Explicitly jump into insert mode immediately
            self.tags_entry.set_position(-1)
            self.ignore_focus_change = False
            return True
        elif key == 'k':
            self.ignore_focus_change = True
            self.skip_tags_entry.grab_focus()
            self.set_mode("INSERT")
            self.skip_tags_entry.set_position(-1)
            self.ignore_focus_change = False
            return True
        elif key == 'a':
            self.ignore_focus_change = True
            self.start_task_entry.grab_focus()
            self.set_mode("INSERT")
            self.start_task_entry.set_position(-1)
            self.ignore_focus_change = False
            return True

        # checkbox toggles
        elif key == 'c':
            self.dry_run_check.set_active(not self.dry_run_check.get_active())
            return True
        elif key == 'd':
            self.diff_check.set_active(not self.diff_check.get_active())
            return True
        elif key == 'D':
            self.debug_check.set_active(not self.debug_check.get_active())
            return True

        # catch-all block
        # guaranteed block of all other typing while parked on an entry in NORMAL mode
        if is_entry and key.isprintable():
            return True

        return False

    def on_launch_clicked(self, button):
        """Executes the generated command."""
        buffer = self.command_view.get_buffer()
        start, end = buffer.get_bounds()
        gui_args = buffer.get_text(start, end, True)

        print(f"Executing api.sh with args: {gui_args}")

        # resolve api.sh path dynamically so it can be called from anywhere
        script_dir = os.path.dirname(os.path.abspath(__file__))
        api_script = os.path.join(script_dir, "api.sh")

        # run api.sh with the selected tags, and then focus on the scratchpad window with the logs
        bash_script = f"""
kitty \\
    --class="dropdown_ansible" \\
    --hold \\
    -o font_size=12 \\
    -o include="$XDG_CONFIG_HOME/kitty/themes/ansible-provision.conf" \\
    bash -c 'exec "{api_script}" {gui_args}' &

# wait for dropdown_ansible to appear in Sway's window tree
MAX_WAIT=50
COUNTER=0
while ! swaymsg -t get_tree | grep -q '"app_id": "dropdown_ansible"'; do
    sleep 0.1
    ((COUNTER++))
    if [[ $COUNTER -ge $MAX_WAIT ]]; then
        echo "Error: Kitty window never appeared."
        exit 1
    fi
done

# Explicitly command Sway to unhide the scratchpad and bring it into focus!
$XDG_CONFIG_HOME/scripts/show-or-launch.sh dropdown_ansible 0.75 0.75
"""

        # start_new_session=True fully detaches the process group so the background loop survives when GUI closes
        subprocess.Popen(bash_script, shell=True, executable='/bin/bash', start_new_session=True)

        # closes the window after launching
        self.close()
