  unsetopt zle              # disable the Zsh Line Editor (interactive features)
  unsetopt prompt_cr        # disable carriage return printing
  unsetopt prompt_subst     # disable prompt substitution
  unfunction precmd         # disable precmd functions (often used by themes)
